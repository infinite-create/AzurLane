namespace = planet_attack

planet_event = {
	id = planet_attack.0
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		if = {
			limit = { 
				OR = {
					is_planet_class = pc_arid
					is_planet_class = pc_alpine
					is_planet_class = pc_arctic
					is_planet_class = pc_continental
					is_planet_class = pc_desert
					is_planet_class = pc_gaia
					is_planet_class = pc_nuked
					is_planet_class = pc_ocean
					is_planet_class = pc_tropical
					is_planet_class = pc_tundra
				}
			}
			change_pc = pc_black_hole
		}
		else = {
			change_pc = pc_gaia
		}
	}
}

planet_event = {
	id = planet_attack.1
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		random_list = {
			10 = {
				change_pc = pc_t_star
				solar_system = {
					every_system_planet = {
						limit = {
							is_star = no
						}
						if = {
							limit = { 
								is_planet_class = pc_habitat
							}
							remove_planet = yes
						}
						else_if = {
							limit = { 
								is_planet_class = pc_ringworld_habitable
							}
							change_pc = pc_ringworld_habitable_damaged
						}
						else_if = {
							limit = { 
								is_planet_class = pc_ringworld_seam
							}
							change_pc = pc_ringworld_seam_damaged
						}
						else_if = {
							limit = { 
								is_planet_class = pc_ringworld_tech
							}
							change_pc = pc_ringworld_tech_damaged
						}
						else_if = {
							limit = { 
								NOR = {
									is_planet_class = pc_ringworld_habitable_damaged
									is_planet_class = pc_ringworld_seam_damaged
									is_planet_class = pc_ringworld_tech_damaged
									is_planet_class = pc_frozen
								}
							}
							change_pc = pc_frozen
						}
					}
				}
			}
			10 = {
				change_pc = pc_neutron_star
				solar_system = {
					every_system_planet = {
						limit = {
							is_star = no
						}
						if = {
							limit = { 
								OR = {
									is_planet_class = pc_habitat
									is_planet_class = pc_asteroid
									is_planet_class = pc_crystal_asteroid
								}
							}
							remove_planet = yes
						}
						else_if = {
							limit = { 
								is_planet_class = pc_ringworld_habitable
							}
							change_pc = pc_ringworld_habitable_damaged
						}
						else_if = {
							limit = { 
								is_planet_class = pc_ringworld_seam
							}
							change_pc = pc_ringworld_seam_damaged
						}
						else_if = {
							limit = { 
								is_planet_class = pc_ringworld_tech
							}
							change_pc = pc_ringworld_tech_damaged
						}
						else_if = {
							limit = { 
								NOR = {
									is_planet_class = pc_ringworld_habitable_damaged
									is_planet_class = pc_ringworld_seam_damaged
									is_planet_class = pc_ringworld_tech_damaged
									is_planet_class = pc_molten
								}
							}
							change_pc = pc_molten
						}
					}
					every_fleet_in_system = {
						limit = {
							NOT = {
								is_ship_class = shipclass_colonizer
							}
						}
						add_modifier = {
							modifier = excessive_radiation
						}
						if = {
							limit = { 
								exists = leader
							}
							leader = {
								kill_leader = {
									show_notification = yes
								}
							}
						}
					}
					add_modifier = {
						modifier = super_moment_of_torsion
						days = 180
					}
				}
				planet_event = {
					id = planet_attack.2
					days = 90
					random = 90
				}
			}
		}
	}
}

planet_event = {
	id = planet_attack.2
	is_triggered_only = yes
	hide_window = yes

	immediate = {
		solar_system = {
			every_system_planet = {
				limit = {
					is_star = no
				}
				every_system_megastructure = {
					delete_megastructure = this
				}
				if = {
					limit = {
						NOR = {
							is_planet_class = pc_frozen
							is_planet_class = pc_gas_giant
						}
					}
					change_pc = pc_frozen
				}
			}
			every_fleet_in_system = {
				limit = {
					NOT = {
						is_ship_class = shipclass_colonizer
					}
				}
				destroy_fleet = this
			}
		}
	}
}
